\newcommand{\deuxSecantes}{
% quatre points A, B, C, D sur un cercle de centre O doivent auparavant être définis
\tkzDefPoint(0,0){O}
\tkzDrawCircle(O,A)
\tkzInterLL(B,C)(A,D)\tkzGetPoint{P}
\tkzDrawSegments(P,A P,B)
\tkzDrawPolygon[very thick](A,B,C,D)
}
